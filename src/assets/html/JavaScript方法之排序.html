冒泡排序

  1. 比较相邻的两个元素，如果其中一个比后一个大，则交换位置
  2. 第一轮的时候最后一个原生应该是最大的
  3. 完成一轮比较后因为最后一个是最大的，所以不用再进行比较，所以只需要进行ary.length-1次循环即可

  4. 4 if (ary[j] > ary[j + 1]) { 
  5. 5 var cur = ary[j]; 
  6. 6 ary[j] = ary[j + 1]; 
  7. 7 ary[j + 1] = cur 
  8. 8 } 
  9. 9 } 
  10. 10 } 
  11. 11 return ary 
  12. 12 }

快速排序

  1. 找基准一个基准值pivot（因为数组是无序的，所以用数组中的哪个都一样。）
  2. 新建两个数组(min,max)遍历数组，以从小到大排序为例，比基准值小的放min中，比基准值大的放max中。
  3. 递归，递归的时候分别对min，max进行操作并在中间concat上基准值，防止死循环

  20. 20 } 
  21. 21 }

插入排序

  1. 从第一个元素开始，该元素可以认为已经被排序
  2. 取出下一个元素，在已经排序的元素序列中从后向前扫描
  3. 如果该元素（已排序）大于新元素，将该元素移到下一位置
  4. 重复步骤3，直到找到已排序的元素小于或者等于新元素的位置
  5. 将新元素插入到下一位置中
  6. 重复步骤2
  1. 1 function insertionSort(ary) { 
  2. 2 var preIndex, current; 
  3. 3 for (var i = 1; i < ary.length; i++) { 
  4. 4 preIndex = i - 1; 
  5. 5 current = ary[i]; 
  6. 6 while(preIndex >= 0 && ary[preIndex] > current) { 
  7. 7 ary[preIndex+1] = ary[preIndex]; 
  8. 8 preIndex--; 
  9. 9 } 
  10. 10 ary[preIndex+1] = current; 
  11. 11 } 
  12. 12 return ary; 
  13. 13 } 

希尔排序

希尔排序是插入排序的一种更高效率的实现。它与插入排序的不同之处在于，它会优先比较距离较远的元素。希尔排序的核心在于间隔序列的设定。既可以提前设定好间隔序列，也可以动态的

  1. 1 function shellSort(ary) { 
  2. 2 var len = ary.length, 
  3. 3 temp, 
  4. 4 gap = 1; 
  5. 5 while (gap < len / 3) { //动态定义间隔序列 
  6. 6 gap = gap * 3 + 1; 
  7. 7 } 
  8. 8 for (gap; gap > 0; gap = Math.floor(gap / 3)) { 
  9. 9 for (var i = gap; i < len; i++) { 
  10. 10 temp = ary[i]; 
  11. 11 for (var j = i - gap; j > 0 && ary[j] > temp; j -= gap) { 
  12. 12 ary[j + gap] = ary[j]; 
  13. 13 } 
  14. 14 ary[j + gap] = temp; 
  15. 15 } 
  16. 16 } 
  17. 17 return ary; 
  18. 18 } 

归并排序

作为一种典型的分而治之思想的算法应用，归并排序的实现由两种方法：

  1. 自上而下的递归（所有递归的方法都可以用迭代重写，所以就有了第2种方法）
  2. 自下而上的迭代
    在《数据结构与算法JavaScript描述》中，作者给出了自下而上的迭代方法。但是对于递归法，作者却认为：

However, it is not possible to do so in JavaScript, as the recursion goes too deep
for the language to handle.
然而,在 JavaScript 中这种方式不太可行,因为这个算法的递归深度对它来讲太深了。

在《JavaScript语言精粹》的第四章里提到了递归问题。对之前的疑问进行了解答：

Some languages offer the tail recursion optimization. This means that if a function returns the result of invoking itself recursively, then the invocation is replaced with a loop, which can significantly speed things up. Unfortunately, JavaScript does not currently provide tail recursion optimization. Functions that recurse very deeply can fail by exhausting the return stack.
一些语言提供了尾递归优化。这意味着如果一个函数返回自身递归调用的结果，那么调用的过程会被替换为一个循环，它可以显著提高速度。遗憾的是，JavaScript当前并没有提供尾递归优化。深度递归的函数可能会因为堆栈溢出而运行失败。

简而言之，就是JavaScript没有对递归进行优化。运用递归函数不仅没有运行速度上的优势，还可能造成程序运行失败。因此不建议使用递归。

和选择排序一样，归并排序的性能不受输入数据的影响，但表现比选择排序好的多，因为始终都是O(n log n）的时间复杂度。代价是需要额外的内存空间。

  1. 1 function mergeSort(ary) { //采用自上而下的递归方法 
  2. 2 var len = ary.length; 
  3. 3 if (len < 2) { 
  4. 4 return ary; 
  5. 5 } 
  6. 6 var middle = Math.floor(len / 2), 
  7. 7 left = ary.slice(0, middle), 
  8. 8 right = ary.slice(middle); 
  9. 9 return merge(mergeSort(left), mergeSort(right)); 
  10. 10 } 
  11. 11 function merge(left, right) { 
  12. 12 var result = []; 
  13. 13 while (left.length > 0 && right.length > 0) { 
  14. 14 if (left[0] <= right[0]) { 
  15. 15 result.push(left.shift()); 
  16. 16 } else { 
  17. 17 result.push(right.shift()); 
  18. 18 } 
  19. 19 } 
  20. 20 while (left.length) 
  21. 21 result.push(left.shift()); 
  22. 22 while (right.length) 
  23. 23 result.push(right.shift()); 
  24. 24 return result; 
  25. 25 } 

堆排序

堆排序可以说是一种利用堆的概念来排序的选择排序。分为两种方法：

  1. 大顶堆：每个节点的值都大于或等于其子节点的值，在堆排序算法中用于升序排列
  2. 小顶堆：每个节点的值都小于或等于其子节点的值，在堆排序算法中用于降序排列
  1. 1 var len; //因为声明的多个函数都需要数据长度，所以把len设置成为全局变量 
  2. 2 function buildMaxHeap(arr) { //建立大顶堆 
  3. 3 len = arr.length; 
  4. 4 for (var i = Math.floor(len / 2); i &gt;= 0; i--) { 
  5. 5 heapify(arr, i); 
  6. 6 } 
  7. 7 } 
  8. 8 function heapify(arr, i) { //堆调整 
  9. 9 var left = 2 * i + 1, 
  10. 10 right = 2 * i + 2, 
  11. 11 largest = i; 
  12. 12 if (left < len && arr[left] > arr[largest]) { 
  13. 13 largest = left; 
  14. 14 } 
  15. 15 if (right < len && arr[right] > arr[largest]) { 
  16. 16 largest = right; 
  17. 17 } 
  18. 18 if (largest != i) { 
  19. 19 swap(arr, i, largest); 
  20. 20 heapify(arr, largest); 
  21. 21 } 
  22. 22 } 
  23. 23 function swap(arr, i, j) { 
  24. 24 var temp = arr[i]; 
  25. 25 arr[i] = arr[j]; 
  26. 26 arr[j] = temp; 
  27. 27 } 
  28. 28 function heapSort(arr) { 
  29. 29 buildMaxHeap(arr); 
  30. 30 for (var i = arr.length - 1; i > 0; i--) { 
  31. 31 swap(arr, 0, i); 
  32. 32 len--; 
  33. 33 heapify(arr, 0); 
  34. 34 } 
  35. 35 return arr; 
  36. 36 } 

计数排序

计数排序的核心在于将输入的数据值转化为键存储在额外开辟的数组空间中。
作为一种线性时间复杂度的排序，计数排序要求输入的数据必须是有确定范围的整数。

  1. 1 function countingSort(ary, maxValue) { 
  2. 2 var bucket = new Array(maxValue+1), 
  3. 3 sortedIndex = 0; 
  4. 4 arrLen = ary.length, 
  5. 5 bucketLen = maxValue + 1; 
  6. 6 for (var i = 0; i < arrLen; i++) { 
  7. 7 if (!bucket[ary[i]]) { 
  8. 8 bucket[ary[i]] = 0; 
  9. 9 } 
  10. 10 bucket[ary[i]]++; 
  11. 11 } 
  12. 12 for (var j = 0; j < bucketLen; j++) { 
  13. 13 while(bucket[j] > 0) { 
  14. 14 ary[sortedIndex++] = j; 
  15. 15 bucket[j]--; 
  16. 16 } 
  17. 17 } 
  18. 18 return ary; 
  19. 19 } 

桶排序

桶排序是计数排序的升级版。它利用了函数的映射关系，高效与否的关键就在于这个映射函数的确定。
为了使桶排序更加高效，我们需要做到这两点：

  1. 在额外空间充足的情况下，尽量增大桶的数量
  2. 使用的映射函数能够将输入的N个数据均匀的分配到K个桶中

同时，对于桶中元素的排序，选择何种比较排序算法对于性能的影响至关重要。

什么时候最快（Best Cases）：
当输入的数据可以均匀的分配到每一个桶中

什么时候最慢（Worst Cases）：
当输入的数据被分配到了同一个桶中

  1. 1 function bucketSort(arr, bucketSize) { 
  2. 2 if (arr.length === 0) { 
  3. 3 return arr; 
  4. 4 } 
  5. 5 var i; 
  6. 6 var minValue = arr[0]; 
  7. 7 var maxValue = arr[0]; 
  8. 8 for (i = 1; i < arr.length; i++) { 
  9. 9 if (arr[i] < minValue) { 
  10. 10 minValue = arr[i]; //输入数据的最小值 
  11. 11 } else if (arr[i] > maxValue) { 
  12. 12 maxValue = arr[i]; //输入数据的最大值 
  13. 13 } 
  14. 14 } 
  15. 15 //桶的初始化 
  16. 16 var DEFAULT_BUCKET_SIZE = 5; //设置桶的默认数量为5 
  17. 17 bucketSize = bucketSize || DEFAULT_BUCKET_SIZE; 
  18. 18 var bucketCount = Math.floor((maxValue - minValue) / bucketSize) + 1; 
  19. 19 var buckets = new Array(bucketCount); 
  20. 20 for (i = 0; i < buckets.length; i++) { 
  21. 21 buckets[i] = []; 
  22. 22 } 
  23. 23 //利用映射函数将数据分配到各个桶中 
  24. 24 for (i = 0; i < arr.length; i++) { 
  25. 25 buckets[Math.floor((arr[i] - minValue) / bucketSize)].push(arr[i]); 
  26. 26 } 
  27. 27 arr.length = 0; 
  28. 28 for (i = 0; i < buckets.length; i++) { 
  29. 29 insertionSort(buckets[i]); //对每个桶进行排序，这里使用了插入排序 
  30. 30 for (var j = 0; j < buckets[i].length; j++) { 
  31. 31 arr.push(buckets[i][j]); 
  32. 32 } 
  33. 33 } 
  34. 34 return arr; 
  35. 35 } 

基数排序

基数排序有两种方法：

  1. MSD 从高位开始进行排序
  2. LSD 从低位开始进行排序

基数排序 vs 计数排序 vs 桶排序
这三种排序算法都利用了桶的概念，但对桶的使用方法上有明显差异：
基数排序：根据键值的每位数字来分配桶
计数排序：每个桶只存储单一键值
桶排序：每个桶存储一定范围的数值
